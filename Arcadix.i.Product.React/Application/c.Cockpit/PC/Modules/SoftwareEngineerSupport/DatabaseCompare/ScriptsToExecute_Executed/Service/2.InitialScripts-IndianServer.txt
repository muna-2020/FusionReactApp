    CREATE procEDURE [dbo].[p_Extranet_Suggestion_GetData]    
(                                  
  @iMainClientId INT,                  
  @iApplicationTypeId INT,                                  
 @vPrimaryKeyValue nvarchar(max)                               
)                 
/*******************************************************************************                              
 Created By: Rohit                 
 Created Date: 07.02.2019                        
 Description:                                   
                              
  Modified By:  Amit Kumar                                
  Modified Date: 05-08-2020                                  
  Description: Common sp change (for Batch preload)      
      
  Modified By: Arvind K                                         
  Modified Date: 24/08/2020                                       
  Description: Included column uModifiedByUserId              
                                              
*************************************************************************************************/         
AS              
BEGIN      
-----------Common approach to fetch data in Batch--------------          
Declare @uSuggestionInformationId uniqueidentifier               
If(@vPrimaryKeyValue = '0')          
BEGIN          
 Set @uSuggestionInformationId = '00000000-0000-0000-0000-000000000000'          
END          
Else          
Begin          
 Set @uSuggestionInformationId = Cast(@vPrimaryKeyValue as uniqueidentifier)

---------------------------------------------------------------        
                               
 If(@iApplicationTypeId=-1)                
 BEGIN      
 select top(5000)(              
  Select (                                  
    Select                
  S.uSuggestionInformationId,                       
  S.uUserId,

  S.iMainClientId,                
  S.tSuggestionText,                
  S.iApplicationTypeId,

  S.cIsMailSent,        
  S.vType,        
  S.iTaskId,

   ,ISNULL((                              
    Select  SA.uSuggestionAttachmentId,SA.uSuggestionInformationId,SA.vFileName,SA.vFileType                             
    From t_Testdrive_Member_Suggestion_Attachments SA                          
    Where S.uSuggestionInformationId =SA.uSuggestionInformationId for json path, include_null_values                          
   ),'[]')   [t_Testdrive_Member_Suggestion_Attachments]                       
    for json path, include_null_values, without_array_wrapper)) as jsonresult                                  
    from t_Testdrive_Member_Suggestion S                 
    where S.iMainClientId=@iMainClientId and s.cIsDeleted= 'N'  and S.uSuggestionInformationId > @uSuggestionInformationId      
 order by S.uSuggestionInformationId          
 END                
 ELSE                
   BEGIN      
   select top(5000)(              
       Select (                                  
    Select                
  S.uSuggestionInformationId,                
  S.iMainClientId,                
  S.uUserId,    
  S.uModifiedByUserId,                
  S.tSuggestionText,                
  S.iApplicationTypeId,                
  S.dtCreatedOn,                
  S.cIsMailSent,        
  S.vType,        
  S.iTaskId,       
   S.dtModifiedOn        
   ,ISNULL((                              
    Select  SA.uSuggestionAttachmentId,SA.uSuggestionInformationId,SA.vFileName,SA.vFileType                             
    From t_Testdrive_Member_Suggestion_Attachments SA                          
    Where S.uSuggestionInformationId =SA.uSuggestionInformationId for json path, include_null_values                          
   ),'[]')   [t_Testdrive_Member_Suggestion_Attachments]                       
    for json path, include_null_values, without_array_wrapper)) as jsonresult                                  
    from t_Testdrive_Member_Suggestion S   
    where S.iMainClientId=@iMainClientId and s.cIsDeleted= 'N' and s.iApplicationTypeId = @iApplicationTypeId   and S.uSuggestionInformationId > @uSuggestionInformationId      
 order by S.uSuggestionInformationId                   
   END                               
END

CREATE procEDURE [dbo].[p_Extranet_Suggestion_EditData]              
(              
  @iMainClientId INT,                   
  @vEditData nvarchar(max)                                   
)                     
/*******************************************************************************                        
 Created By: Rohit Kumar           
 Created Date: 08-02-2019                  
 Description: This edit the Suggestion data                               
                        
  Modified By: Arvind K                                         
  Modified Date: 24/08/2020                                       
  Description: Included column uModifiedByUserId                                    
                         
 ********************************************************************************/                                       
As                                      
BEGIN                                      
    declare @uSuggestionInformationId uniqueIdentifier                
  --Update Main Table                  
   select @uSuggestionInformationId = uSuggestionInformationId from OPENJSON(@vEditData) with (uSuggestionInformationId uniqueidentifier)                 
   UPDATE S                    
   set                     
   S.iApplicationTypeId = SJson.iApplicationTypeId,                    
   S.uModifiedByUserId =  SJson.uUserId,                                           
   S.iMainClientId=@iMainClientId,      
   S.tSuggestionText = SJson.tSuggestionText,      
   S.iTaskId = SJSon.iTaskId,

   from t_Testdrive_Member_Suggestion S,OPENJSON(@vEditData)                                
 with (                    
      iApplicationTypeId int,                
   uUserId uniqueIdentifier,                          
  tSuggestionText nvarchar(max),      
  iTaskId int,      
  vType nvarchar(50)      
  ) SJson                                
  where uSuggestionInformationId = @uSuggestionInformationId                  
                  
   --Adding subTable data                  
 declare @vDataJson nvarchar(max)                                      
declare @uSuggestionAttachmentId uniqueidentifier                                      
declare Cur_SuggestionData cursor                                
                                    
 For Select value from OPENJSON (@vEditData, '$.t_Testdrive_Member_Suggestion_Attachments')                              
 open Cur_SuggestionData                            
 Fetch next from Cur_SuggestionData into @vDataJson                 
 WHILE(@@FETCH_STATUS<>-1)                            
 BEGIN          
 SELECT @uSuggestionAttachmentId = uSuggestionAttachmentId from OPENJSON (@vDataJson) with(uSuggestionAttachmentId UNIQUEIDENTIFIER)              
 if (@uSuggestionAttachmentId IS NOT NULL AND Exists(select * from  t_Testdrive_Member_Suggestion_Attachments where uSuggestionAttachmentId = @uSuggestionAttachmentId)   )                  
  BEGIN                  
       update SA Set                  
SA.vFileName=PVJson.vFileName,                  
SA.vFileType = PVJson.vFileType              
from t_Testdrive_Member_Suggestion_Attachments SA,OPENJSON(@vDataJson)                                                
with (                            
vFileName nvarchar(200),                                                          
vFileType nvarchar(50)                                       
) PVJson              
where SA.uSuggestionInformationId = @uSuggestionInformationId AND SA.uSuggestionAttachmentId = @uSuggestionAttachmentId                                      
END                                      
ELSE                                      
BEGIN                                      
insert into  t_testdrive_Member_Suggestion_Attachments(                                      
uSuggestionInformationId,          
vFileName,                
vFileType ,                
uSuggestionAttachmentId                                    
)                 
select   
@uSuggestionInformationId,          
vFileName,                
vFileType ,                                        
NEWID()                                             
from OPENJSON(@vDataJson)                                        
with (              
    vFileName nvarchar(200),                
  vFileType   nvarchar(50)          
)              
                                  END                                      
                                             
Fetch next from Cur_SuggestionData into @vDataJson                                      
                                                
END                                          
close Cur_SuggestionData                                          
deallocate Cur_SuggestionData                                      
                                      
                                      
--Return the data                  
 Select (                                      
    Select                    
  S.uSuggestionInformationId,                    
  S.iApplicationTypeId,                    
  S.iMainClientId,                    
  S.tSuggestionText,                    
  S.uUserId,     
  S.uModifiedByUserId,                    
  S.iTaskId,      
  S.vType ,      
  S.dtCreatedOn,      
  S.dtModifiedOn,    
  S.cIsMailSent                           
   ,ISNULL((          
    Select  SA.uSuggestionInformationId, SA.vFileName,SA.vFileType, SA.uSuggestionAttachmentId          
    From t_Testdrive_Member_Suggestion_Attachments SA                              
    Where S.uSuggestionInformationId =SA.uSuggestionInformationId for json path, include_null_values                              
   ),'[]')   [t_Testdrive_Member_Suggestion_Attachments]                           
    for json path, include_null_values, without_array_wrapper) as jsonresult                                      
    from t_Testdrive_Member_Suggestion S                     
    where S.iMainClientId = @iMainClientId  and uSuggestionInformationId = @uSuggestionInformationId                   
   END

CREATE procEDURE [dbo].[p_Extranet_OpenApplicationCredential_GetGateKeeperUrl]               
(                
 @iMainClientId int,                
 @vTargetType nvarchar(max),        
 @vGroup nvarchar(max) ,        
 @iLanguageId int               
)                
AS                
Begin                  
/********************************************************************************************                  
Created By: Amit Kumar                  
Created Date: 07.05.2020                  
Description:            
        
 Modified By:  Amit Kumar                                      
 Modified Date:   08.06.2020                                      
 Description:  Added two more filters @vGroup,@iLanguageId         
    
 Modified By:   Arvind K                     
Modified Date:  01.09.2020        
Description:  Modified proc to return data based on cIsForFusion      
              
********************************************************************************************/                  
    Declare @uTestURLId uniqueidentifier,                
        @iNoOfTimeShown int,                
        @iNumberOfTimeToBeShown int,                
  @vCurrentURL nvarchar(max),                
  @vQueryString nvarchar(max),              
  @vApplicationName nvarchar(max)                
                
      
   -- insert into t_framework_exception values(newid(),@iLanguageId,@iMainClientId,'',@vTargetType,@vGroup,'GatekeeperLog',GETDATE())      
              
 if Exists ( Select  uApplicationServerId                
                from    t_Framework_ApplicationServers                
                where   cIsCurrentURL = 'Y' and cIsActive='Y' and cIsDeleted='N' and cIsForFusion = 'Y'          
                        and Isnull(iNumberOfTimeShown,0) < iNumberOfTimeToBeShown and iMainClientId=@iMainClientId and Ltrim(RTrim(vTargetType)) = @vTargetType and Ltrim(RTrim(vGroup)) = @vGroup and iLanguageId =@iLanguageId  )                   
        Begin                  
                
            Select top 1                
     @uTestURLId = uApplicationServerId,                
                    @iNoOfTimeShown = Isnull(iNumberOfTimeShown,0),                
                    @iNumberOfTimeToBeShown = iNumberOfTimeToBeShown,                
     @vCurrentURL = vHostName, @vQueryString=vQueryString ,@vApplicationName=vApplicationName              
            from    t_Framework_ApplicationServers                
            where   cIsCurrentURL = 'Y' and cIsActive='Y' and cIsDeleted='N'  and cIsForFusion = 'Y'               
                    and Isnull(iNumberOfTimeShown,0) < iNumberOfTimeToBeShown  and iMainClientId=@iMainClientId                
      and Ltrim(RTrim(vTargetType)) = @vTargetType   and Ltrim(RTrim(vGroup)) = @vGroup  and iLanguageId =@iLanguageId           
   order by iOrder                
            print @iNoOfTimeShown                
--Condition 2                   
            Update  t_Framework_ApplicationServers                
            set     iNumberOfTimeShown = @iNoOfTimeShown + 1,iUrlDisplayCounter = isnull(iUrlDisplayCounter,0)+1                
            where   cIsCurrentURL = 'Y' and cIsActive='Y'   and cIsDeleted='N'  and cIsForFusion = 'Y'              
                    and  uApplicationServerId = @uTestURLId and iMainClientId=@iMainClientId and Ltrim(RTrim(vTargetType)) = @vTargetType   and Ltrim(RTrim(vGroup)) = @vGroup  and iLanguageId =@iLanguageId           
                               
        End                  
    Else                 
        Begin                  
--Condition 1                   
            if Exists ( Select uApplicationServerId                
                        from    t_Framework_ApplicationServers                
                        where   cIsCurrentURL = 'N' and iMainClientId=@iMainClientId and cIsActive='Y' and cIsDeleted='N' and Ltrim(RTrim(vTargetType)) = @vTargetType     
        and Ltrim(RTrim(vGroup)) = @vGroup and iLanguageId =@iLanguageId  and cIsForFusion = 'Y' )                 
                Begin                  
                    Select top 1                
        @uTestURLId = uApplicationServerId,                
                            @vCurrentURL = vHostName,                
                            @iNoOfTimeShown = Isnull(iNumberOfTimeShown,0), @vQueryString=vQueryString ,@vApplicationName=vApplicationName               
                    from    t_Framework_ApplicationServers                
                    where   cIsCurrentURL = 'N' and iMainClientId=@iMainClientId and cIsActive='Y'  and cIsDeleted='N'  and cIsForFusion = 'Y'                
        and Ltrim(RTrim(vTargetType)) = @vTargetType   and Ltrim(RTrim(vGroup)) = @vGroup    and iLanguageId =@iLanguageId         
       order by iOrder                
                
             Update  t_Framework_ApplicationServers                
                    set     cIsCurrentURL = 'Y',                
                            iNumberOfTimeShown = @iNoOfTimeShown + 1                
       ,iUrlDisplayCounter = isnull(iUrlDisplayCounter,0)+1                
                    where   uApplicationServerId = @uTestURLId  and iMainClientId=@iMainClientId and cIsActive='Y' and cIsDeleted='N'   and cIsForFusion = 'Y'              
        and Ltrim(RTrim(vTargetType)) = @vTargetType      and Ltrim(RTrim(vGroup)) = @vGroup  and iLanguageId =@iLanguageId        
                End                  
            Else                 
                Begin                  
--Conditin 3 Reset                  
                    Update  t_Framework_ApplicationServers                
                    set     cIsCurrentURL = 'N',                
                            iNumberOfTimeShown = 0  where iMainClientId=@iMainClientId and cIsActive='Y'  and cIsDeleted='N'  and cIsForFusion = 'Y'              
        and Ltrim(RTrim(vTargetType)) = @vTargetType       and Ltrim(RTrim(vGroup)) = @vGroup and iLanguageId =@iLanguageId        
                    Select top 1                
                @uTestURLId = uApplicationServerId,                
                            @vCurrentURL = vHostName,                
                            @iNoOfTimeShown = iNumberOfTimeShown, @vQueryString=vQueryString ,@vApplicationName=vApplicationName               
                    from    t_Framework_ApplicationServers                
                    where   cIsCurrentURL = 'N' and iMainClientId=@iMainClientId and cIsActive='Y' and cIsDeleted='N'   and cIsForFusion = 'Y'                
        and Ltrim(RTrim(vTargetType)) = @vTargetType       and Ltrim(RTrim(vGroup)) = @vGroup and iLanguageId =@iLanguageId        
       order by iOrder                  
                    Update  t_Framework_ApplicationServers                
                    set     cIsCurrentURL = 'Y',                
                            iNumberOfTimeShown = @iNoOfTimeShown + 1                
       ,iUrlDisplayCounter = isnull(iUrlDisplayCounter,0)+1                
                    where   uApplicationServerId = @uTestURLId  and iMainClientId=@iMainClientId and cIsActive='Y'  and cIsDeleted='N'  and cIsForFusion = 'Y'              
       and Ltrim(RTrim(vTargetType)) = @vTargetType       and Ltrim(RTrim(vGroup)) = @vGroup and iLanguageId =@iLanguageId        
                End                   
        End                  
    SELECT(            
    Select  @vCurrentURL  vCurrentURL, @vQueryString vQueryString ,@vApplicationName vApplicationName              
 for json path, INCLUDE_NULL_VALUES, without_array_wrapper)                      
     as JSONResult            
End
